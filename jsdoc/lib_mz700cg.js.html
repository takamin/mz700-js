    "use strict";

/**
 * MZ-700 Character Generator.
 *
 * @constructor
 *
 * @param {number[]} patternBuffer
 * character pattern memory.
 *
 * @param {number} width
 * Pixel width of a character.
 *
 * @param {number} height
 * Pixel height of a character.
 */
function mz700cg(patternBuffer, width, height) {

    patternBuffer = patternBuffer || mz700cg.ROM;
    width = width || 8;
    height = height || 8;

    this._fontTable = null;
    this._patternBuffer = patternBuffer;
    this._width = width;
    this._height = height;

    this.createFontTable();
}

/**
 * Create font table.
 *
 * @returns {undefined}
 */
mz700cg.prototype.createFontTable = function() {
    // 256 character x 2 bank x 8 background colors x 8 foreground colors.
    this._fontTable = new Array(2 * 256 * 8 * 8);
    for(var atb = 0; atb < 2; atb++) {
        for(var dispCode = 0; dispCode < 256; dispCode++) {
            this.initFont(atb, dispCode);
        }
    }
};

/**
 * Initialize specific font in the table.
 *
 * @param {number} atb
 * An attribute bit which select charactoer bank by zero or one.
 *
 * @param {number} dispCode
 * A display code.
 *
 * @returns {undefined}
 */
mz700cg.prototype.initFont = function(atb, dispCode) {
    var pattern = this._patternBuffer[atb * 256 + dispCode];
    for(var bg = 0; bg < 8; bg++) {
        for(var fg = 0; fg < 8; fg++) {
            var attr = (atb << 7)|(fg << 4) | bg;
            this._fontTable[mz700cg.tableIndex(attr, dispCode)] =
                new mz700cg.FontImage(pattern,
                    mz700cg.Colors[fg], mz700cg.Colors[bg],
                    this._width, this._height);
        }
    }
};

/**
 * @param {number} atb
 * An attribute bit which select charactoer bank by zero or one.
 *
 * @param {number} dispCode
 * A display code.
 *
 * @param {number} row
 * Index of row to be set the pattern.
 *
 * @param {number} pattern
 * Bit pattern to be set at the row.
 *
 * @returns {undefined}
 */
mz700cg.prototype.setPattern = function(atb, dispCode, row, pattern) {
    var cpos = atb * 256 + dispCode;
    this._patternBuffer[cpos][row] = pattern;
    this.initFont(atb, dispCode);
};

/**
 * Get a index of the font table by attribute and display code
 *
 * @param {number} attr
 * An attribute 8 bits value including atb and fore and background color.
 * [ atb:1 bit ][ fg color index:3 bits ][ bg color index: 3 bits ]
 *
 * @param {number} dispCode
 * A display code.
 *
 * @returns {number} an index of the font table.
 */
mz700cg.tableIndex = function(attr, dispCode) {
    return attr << 8 | dispCode;
};

/**
 * Get a font image instance.
 *
 * @param {number} attr
 * An attribute 8 bits value including atb and fore and background color.
 * [ atb:1 bit ][ fg color index:3 bits ][ bg color index: 3 bits ]
 *
 * @param {number} dispCode
 * A display code.
 *
 * @returns {mz700cg.FontImage} A font image.
 */
mz700cg.prototype.get = function(attr, dispCode) {
    return this._fontTable[mz700cg.tableIndex(attr, dispCode)];
};

/**
 * Font image offering CANVAS element.
 *
 * @param {number[]} pattern
 * A pixel bit pattern of font. An 8 length array of 8 bit numbers.
 *
 * @param {number} fg
 * An index of foreground color (0-7).
 *
 * @param {number} bg
 * An index of foreground color (0-7).
 *
 * @param {number} width
 * Pixel width of a character.
 *
 * @param {number} height
 * Pixel height of a character.
 *
 * @constructor
 */
mz700cg.FontImage = function(pattern, fg, bg, width, height) {
    this.getImageData = function() {
        var canvas = document.createElement("CANVAS");
        canvas.setAttribute("width", width + "px");
        canvas.setAttribute("height", height + "px");
        var ctx = canvas.getContext("2d");
        for(var row = 0; row < 8; row++) {
            var bits = pattern[row];
            for(var col = 0; col < 8; col++) {
                if((bits & (0x80 >> col)) != 0) {
                    ctx.fillStyle = fg;
                } else {
                    ctx.fillStyle = bg;
                }
                ctx.fillRect(col, row, 1, 1);
            }
        }
        this.getImageData = function() {
            return ctx.getImageData(0, 0, width, height);
        };
        return this.getImageData();
    }.bind(this);
};
